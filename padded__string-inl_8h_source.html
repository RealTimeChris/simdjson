15 // The allocate_padded_buffer function is a low-level function to allocate
16 // memory with padding so we can read past the "length" bytes safely. It is used
17 // by the padded_string class automatically. It returns nullptr in case of
18 // error: the caller should check for a null pointer. The length parameter is
19 // the maximum size in bytes of the string. The caller is responsible to free
20 // the memory (e.g., delete[] (...)).
21 inline char *allocate_padded_buffer(size_t length) noexcept {
22  const size_t totalpaddedlength = length + SIMDJSON2_PADDING;
23  if (totalpaddedlength < length) {
24  // overflow
25  return nullptr;
26  }
27 #ifdef FUZZING_BUILD_MODE_UNSAFE_FOR_PRODUCTION
28  // avoid getting out of memory
29  if (totalpaddedlength > (1UL << 20)) {
30  return nullptr;
31  }
32 #endif
33 
34  char *padded_buffer = new (std::nothrow) char[totalpaddedlength];
35  if (padded_buffer == nullptr) {
36  return nullptr;
37  }
38  // We write nulls in the padded region to avoid having uninitialized
39  // content which may trigger warning for some sanitizers
40  std::memset(padded_buffer + length, 0, totalpaddedlength - length);
41  return padded_buffer;
42 } // allocate_padded_buffer()

135 padded_string::load(std::string_view filename) noexcept {
136  // Open the file
137  SIMDJSON2_PUSH_DISABLE_WARNINGS
138  SIMDJSON2_DISABLE_DEPRECATED_WARNING // Disable CRT_SECURE warning on MSVC:
139  // manually verified this is safe
140  std::FILE *fp = std::fopen(
141  filename.data(), "rb");
142  SIMDJSON2_POP_DISABLE_WARNINGS
143 
144  if (fp == nullptr) {
145  return IO_ERROR;
146  }
147 
148  // Get the file size
149  int ret;
150 #if SIMDJSON2_VISUAL_STUDIO && !SIMDJSON2_IS_32BITS
151  ret = _fseeki64(fp, 0, SEEK_END);
152 #else
153  ret = std::fseek(fp, 0, SEEK_END);
154 #endif // _WIN64
155  if (ret < 0) {
156  std::fclose(fp);
157  return IO_ERROR;
158  }
159 #if SIMDJSON2_VISUAL_STUDIO && !SIMDJSON2_IS_32BITS
160  __int64 llen = _ftelli64(fp);
161  if (llen == -1L) {
162  std::fclose(fp);
163  return IO_ERROR;
164  }
165 #else
166  long llen = std::ftell(fp);
167  if ((llen < 0) || (llen == LONG_MAX)) {
168  std::fclose(fp);
169  return IO_ERROR;
170  }
171 #endif
172 
173  // Allocate the padded_string
174  size_t len = static_cast<size_t>(llen);
175  padded_string s(len);
176  if (s.data() == nullptr) {
177  std::fclose(fp);
178  return MEMALLOC;
179  }
180 
181  // Read the padded_string
182  std::rewind(fp);
183  size_t bytes_read = std::fread(s.data(), 1, len, fp);
184  if (std::fclose(fp) != 0 || bytes_read != len) {
185  return IO_ERROR;
186  }
187 
188  return s;
189 }
